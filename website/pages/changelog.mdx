import { useEffect, useState } from 'react'

export function Changelog() {
  const [releases, setReleases] = useState([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState(null)

  useEffect(() => {
    fetch('https://api.github.com/repos/pamasiga/pamasiga-os/releases')
      .then(res => {
        if (!res.ok) {
          throw new Error(`Failed to fetch releases: ${res.status}`)
        }
        return res.json()
      })
      .then(data => {
        // Ensure data is an array before setting state
        setReleases(Array.isArray(data) ? data : [])
        setLoading(false)
      })
      .catch(err => {
        console.error('Error fetching releases:', err)
        setError(err.message)
        setLoading(false)
      })
  }, [])

  if (loading) {
    return <div className="text-center py-8">Loading changelog...</div>
  }

  if (error) {
    return (
      <div className="text-center py-8">
        <p className="text-gray-600 dark:text-gray-400 mb-4">Unable to load releases at this time.</p>
        <p className="text-sm text-gray-500 dark:text-gray-500">Check back later or visit <a href="https://github.com/pamasiga/pamasiga-os/releases" target="_blank" rel="noopener noreferrer" className="text-blue-600 hover:underline">GitHub Releases</a> directly.</p>
      </div>
    )
  }

  return (
    <div className="changelog space-y-8">
      {releases.length === 0 ? (
        <p className="text-gray-600 dark:text-gray-400">No releases yet. Check back soon!</p>
      ) : (
        releases.map((release) => (
          <div key={release.id} className="border-l-4 border-blue-500 pl-6 pb-6">
            <div className="flex items-center gap-3 mb-2">
              <h2 className="text-2xl font-bold">
                <a
                  href={release.html_url}
                  target="_blank"
                  rel="noopener noreferrer"
                  className="hover:text-blue-600 dark:hover:text-blue-400"
                >
                  {release.name || release.tag_name}
                </a>
              </h2>
              <span className="px-3 py-1 bg-blue-100 dark:bg-blue-900 text-blue-700 dark:text-blue-300 rounded-full text-sm font-semibold">
                {release.tag_name}
              </span>
              {release.prerelease && (
                <span className="px-3 py-1 bg-yellow-100 dark:bg-yellow-900 text-yellow-700 dark:text-yellow-300 rounded-full text-sm">
                  Pre-release
                </span>
              )}
            </div>
            <p className="text-sm text-gray-500 dark:text-gray-400 mb-4">
              Released {new Date(release.published_at).toLocaleDateString('en-US', {
                year: 'numeric',
                month: 'long',
                day: 'numeric'
              })}
            </p>
            <div
              className="prose dark:prose-invert max-w-none"
              dangerouslySetInnerHTML={{ __html: release.body_html || release.body }}
            />
          </div>
        ))
      )}
    </div>
  )
}

## Versioning

PAMASIGA OS follows [Semantic Versioning](https://semver.org/):

- **MAJOR** version for incompatible API changes
- **MINOR** version for new functionality in a backward compatible manner
- **PATCH** version for backward compatible bug fixes
